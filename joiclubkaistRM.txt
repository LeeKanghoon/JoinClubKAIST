CREATE TABLE STUDENT
(Sname VARCHAR(30) NOT NULL,
Sid INT NOT NULL,
Major VARCHAR(5),
Minor VARCHAR(5),
Nationality VARCHAR(2),
Gender CHAR(1),
ID VARCHAR(20),
Password VARCHAR(20),
UNIQUE(ID),
PRIMARY KEY (Sid) );

CREATE TABLE CLUB
(Cname VARCHAR(20) NOT NULL,
 Csn INT NOT NULL,
 Class VARCHAR(30),
 District VARCHAR(18),
 Department VARCHAR(25),
 Establish INT,
 Num_member VARCHAR(5),
 Num_recruit VARCHAR(5),
 Activity_time CHAR(15),
 Phone CHAR(13),
 Homepage VARCHAR(50),
 Room VARCHAR(15),
 CSid INT,
 PRIMARY KEY (Csn),
 UNIQUE (Cname),
 FOREIGN KEY (CSid) REFERENCES STUDENT(Sid) );

CREATE TABLE EVENT
(Eno INT NOT NULL,
 Ename VARCHAR(40),
 Edate CHAR(8),
 Stime CHAR(4),
 Etime CHAR(4),
 Loc VARCHAR(20),
 ECsn INT NOT NULL,
 PRIMARY KEY(Eno),
 FOREIGN KEY (ECsn) REFERENCES CLUB(Csn) );



CREATE TABLE BOOKMARK
(BCsn INT NOT NULL,
 BSid INT NOT NULL,
 PRIMARY KEY (BCsn, BSid),
 FOREIGN KEY (BCsn) REFERENCES CLUB(Csn),
 FOREIGN KEY (BSid) REFERENCES STUDENT(Sid) );

CREATE TABLE INTERESTED_IN
(IEno INT NOT NULL,
 ISid INT NOT NULL,
 PRIMARY KEY (IEno, ISid), 
 FOREIGN KEY(IEno) REFERENCES EVENT(Eno),
 FOREIGN KEY(ISid) REFERENCES STUDENT(Sid) );
